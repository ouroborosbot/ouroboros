     4	export type InnerDialogWorkerReason = "boot" | "heartbeat" | "instinct"
     5	
     6	export interface InnerDialogWorkerMessage {
     7	  type: "heartbeat" | "shutdown" | string
     8	}
     9	
    10	export interface InnerDialogWorkerController {
    11	  run(reason: InnerDialogWorkerReason): Promise<void>
    12	  handleMessage(message: unknown): Promise<void>
    13	}
    14	
    15	export function createInnerDialogWorker(
    16	  runTurn: (options: { reason: InnerDialogWorkerReason }) => Promise<unknown> = (options) => runInnerDialogTurn(options),
    17	): InnerDialogWorkerController {
    18	  let running = false
    19	
    20	  async function run(reason: InnerDialogWorkerReason): Promise<void> {
    21	    if (running) return
    22	    running = true
    23	    try {
    24	      await runTurn({ reason })
    25	    } catch (error) {
    26	      emitNervesEvent({
    27	        level: "error",
    28	        component: "senses",
    29	        event: "senses.inner_dialog_worker_error",
    30	        message: "inner dialog worker turn failed",
    31	        meta: {
    32	          reason,
    33	          error: error instanceof Error ? error.message : String(error),
    34	        },
    35	      })
    36	    } finally {
    37	      running = false
    38	    }
    39	  }
    40	
    41	  async function handleMessage(message: unknown): Promise<void> {
    42	    if (!message || typeof message !== "object") return
    43	    const maybeMessage = message as Partial<InnerDialogWorkerMessage>
    44	    if (maybeMessage.type === "heartbeat") {
    45	      await run("heartbeat")
    46	      return
    47	    }
    48	    if (maybeMessage.type === "shutdown") {
    49	      process.exit(0)
    50	    }
    51	  }
    52	
    53	  return { run, handleMessage }
    54	}
    55	
    56	export async function startInnerDialogWorker(): Promise<void> {
    57	  const worker = createInnerDialogWorker()
    58	  process.on("message", (message) => {
    59	    void worker.handleMessage(message)
    60	  })
    61	  process.on("disconnect", () => {
    62	    process.exit(0)
    63	  })
    64	  await worker.run("boot")
    65	}
