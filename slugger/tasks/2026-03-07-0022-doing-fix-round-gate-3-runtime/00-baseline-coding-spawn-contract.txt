    24	function defaultSubagentPath(subagent: CodingSessionRequest["subagent"]): string {
    25	  return path.join(getRepoRoot(), "subagents", `work-${subagent}.md`)

    28	function buildCommandArgs(runner: CodingRunner, workdir: string): { command: string; args: string[] } {
    29	  if (runner === "claude") {
    30	    return {
    31	      command: "claude",
    32	      args: ["-p", "--dangerously-skip-permissions", "--add-dir", workdir],
    33	    }
    34	  }
    35	
    36	  return {
    37	    command: "codex",
    38	    args: ["exec", "--skip-git-repo-check", "--cwd", workdir],
    39	  }
    40	}

    42	function buildPrompt(request: CodingSessionRequest, deps: Required<Pick<SpawnCodingDeps, "existsSync" | "readFileSync">>): string {
    43	  const sections: string[] = []
    44	
    45	  const subagentPath = defaultSubagentPath(request.subagent)
    46	  if (deps.existsSync(subagentPath)) {
    47	    const instructions = deps.readFileSync(subagentPath, "utf-8").trim()
    48	    if (instructions.length > 0) {
    49	      sections.push(instructions)
    50	    }
    51	  }
    52	
    53	  if (request.stateFile && deps.existsSync(request.stateFile)) {
    54	    const stateContent = deps.readFileSync(request.stateFile, "utf-8").trim()
    55	    if (stateContent.length > 0) {
    56	      sections.push(`State file (${request.stateFile}):\n${stateContent}`)
    57	    }
    58	  }
    59	
    60	  sections.push(request.prompt)
    61	  return sections.join("\n\n---\n\n")
    62	}

    64	export function spawnCodingProcess(request: CodingSessionRequest, deps: SpawnCodingDeps = {}): SpawnCodingResult {
    65	  const spawnFn = deps.spawnFn ?? ((command, args, options) => nodeSpawn(command, args, options) as unknown as CodingProcess)
    66	  const existsSync = deps.existsSync ?? fs.existsSync
    67	  const readFileSync = deps.readFileSync ?? fs.readFileSync
    68	
    69	  const prompt = buildPrompt(request, { existsSync, readFileSync })
    70	  const { command, args } = buildCommandArgs(request.runner, request.workdir)
    71	
    72	  emitNervesEvent({
    73	    component: "repertoire",
    74	    event: "repertoire.coding_spawn_start",
    75	    message: "spawning coding session process",
    76	    meta: { runner: request.runner, subagent: request.subagent, workdir: request.workdir },
    77	  })
    78	
    79	  const proc = spawnFn(command, args, {
    80	    cwd: request.workdir,
    81	    stdio: ["pipe", "pipe", "pipe"],
    82	  })
    83	
    84	  proc.stdin.write(`${prompt}\n`)
    85	
    86	  emitNervesEvent({
    87	    component: "repertoire",
    88	    event: "repertoire.coding_spawn_end",
    89	    message: "spawned coding session process",
    90	    meta: { runner: request.runner, subagent: request.subagent, pid: proc.pid ?? null },
    91	  })
    92	
    93	  return { process: proc, command, args, prompt }
    94	}

    10	function requireArg(args: Record<string, string>, key: string): string | null {
    11	  const value = args[key]
    12	  if (!value || value.trim().length === 0) {
    13	    return null
    14	  }
    15	  return value.trim()
    16	}
    17	
    18	function parseRunner(value: string): CodingRunner | null {
    19	  return RUNNERS.includes(value as CodingRunner) ? (value as CodingRunner) : null
    20	}
    21	
    22	function parseSubagent(value: string): CodingSubagent | null {
    23	  return SUBAGENTS.includes(value as CodingSubagent) ? (value as CodingSubagent) : null
    24	}
    25	
    26	function optionalArg(args: Record<string, string>, key: "taskRef" | "scopeFile" | "stateFile"): string | undefined {
    27	  const raw = args[key]
    28	  if (!raw) return undefined
    29	  const trimmed = raw.trim()
    30	  return trimmed.length > 0 ? trimmed : undefined
    31	}
